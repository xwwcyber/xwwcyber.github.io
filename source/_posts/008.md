**mysql -uroot -p 登录数据库**

show databases;查看数据库

create database Dname;创建数据库

drop database Dname;删除数据库

show create database Dname;

alter database Dname defalut character set CharacterSetName;修改数据库

use Dname;使用数据库



show tables；查看一个数据库中有多少张表

create table Tname；创建表

show create table Tname;/desc Tname；查看表

alter table Tname;删除表





Alter table tName add field datatype；添加列(字段)

alter table tName change oldfield  newfield datatype；对字段名进行修改

Alter table tName modify field newDataType;对字段名的数据类型【约束】进行修改

alter table Tname drop tname;删除表中数据

insert into Tname values（）；添加表中数据

update Tname set tname=newname where ....;更新数据

create table newTname like Tname；复制表结构，不复制表的数据

create table newTname select *from Tname；复制表结构和表的数据

delete from Tname where......;删除符合条件的数据（只删除数据，不删除表结构）

truncate Tname；删除表中所有数据；





alter table Tname add primary key（tname）；单个主键

alter table Tname add primary key（tname1，tname2）；联合主键，需要两个完全相同才算重复



alter table Tname modify tname type auto_increment；设置自动增加

  

alter table Tname modify tname type；

alter table Tname drop primary key；



alter table Tname modify tname type default ....;设置默认值
